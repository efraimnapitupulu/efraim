// A. Seleksi Elemen
// Disarankan menggunakan const untuk elemen yang tidak akan diubah referensinya.
const navItems = document.querySelectorAll('#navMenu li');
const pages = document.querySelectorAll('.page');
const techBtns = document.querySelectorAll('.tech-btn');
const techDescInner = document.getElementById('techDescInner');

// B. Data Deskripsi Teknologi
const techDesc = {
  html: `HTML5 adalah bahasa markup utama untuk membangun **struktur dasar website**. HTML digunakan untuk membuat elemen dan kerangka halaman seperti teks, gambar, link, dan tabel.`,
  css: `CSS3 digunakan untuk memperindah tampilan website dengan **style, warna, animasi, dan layout modern** agar website lebih menarik, responsif, dan konsisten.`,
  js: `JavaScript adalah bahasa pemrograman agar website menjadi **interaktif dan dinamis**, seperti membuat animasi, validasi form, serta mengelola data pada halaman web.`
};

// --- FUNGSI UTAMA 1: Navigasi Page (SPA-style) ---
navItems.forEach(li => {
  li.addEventListener('click', function() {
    // 1. Dapatkan nama menu yang dituju
    const targetMenu = this.dataset.menu; 

    // 2. Hapus kelas 'active' dari semua item navigasi dan halaman
    // Menggunakan perulangan (forEach) yang terpisah lebih jelas untuk tujuan ini.
    navItems.forEach(x => x.classList.remove('active'));
    pages.forEach(p => p.classList.remove('active'));
    
    // 3. Tambahkan kelas 'active' ke item navigasi yang diklik dan halaman target
    this.classList.add('active'); // Menggunakan 'this' untuk elemen yang diklik
    document.querySelector(`.page-${targetMenu}`).classList.add('active');

    // 4. Reset Deskripsi Teknologi (jika pindah dari Technology)
    if (targetMenu !== "technology") {
      // Menggunakan textContent atau innerHTML kosong untuk membersihkan konten
      techDescInner.innerHTML = ""; 
      
      // Reset semua tombol teknologi
      techBtns.forEach(b => b.classList.remove('active'));
    }
  });
});

// --- FUNGSI UTAMA 2: Penjelasan Teknologi Interaktif ---
techBtns.forEach(btn => {
  btn.addEventListener('click', function() {
    // 1. Dapatkan nama teknologi dari data attribute
    const tech = this.dataset.tech; // Menggunakan 'this' dan dataset

    // 2. Hapus status 'active' dari semua tombol, lalu aktifkan tombol yang diklik
    techBtns.forEach(b => b.classList.remove('active'));
    this.classList.add('active');

    // 3. Masukkan deskripsi.
    // Perbaikan KRITIS: Anda salah mengetik 'innerHML' menjadi 'innerHTML'.
    // Saya menggunakan 'innerHTML' di sini untuk memanfaatkan Markdown (bold) di data deskripsi.
    techDescInner.innerHTML = techDesc[tech];
  });
});
